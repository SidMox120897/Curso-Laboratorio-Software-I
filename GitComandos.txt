git init	//Crear Git nuevo
git add <file>	//pasar archivos del working directory al staging area
git add .	//Agrega todos los archivos de trabajo
git status	//para ver en que estado estan tus archivos
git commit	//para pasar del staging area al repositorio (crear una foto version del codigo)
git commit -m "Mensaje" //crea un commint con su respectivo mensaje
git push	//para subirlo a un repositorio remoto
git pull	//para cargar los cambios de otros desarrolladores
git clone	//para crear una copia de un servidor a tu PC
git log 	//muestra todos lss diferentes commit o snapshot
git checkout -- <file>	// regresa al anterior commit del archivo seleccionado
git diff <file>		//mira las diferencias entre las versiones de los archivos

.gitignore		//es un archivo q permite ignorar archivos

git branch 	//Muestra los proyectos
git branch <nombre-Proyecto>	//crea una nueva version
git checkout <nombre-Proyecto> 	//manda al Branch nombrado
